country_event = {
	id = ORA.114514
	is_triggered_only = yes
	hidden = yes

	immediate = {
		if = {
			limit = {
				has_country_flag = {
					flag = ORA_rad
					value > 2
				}
			}
			country_event = ORA.9
		}
		if = {
			limit = {
				has_country_flag = {
					flag = ORA_mid
					value > 2
				}
			}
			country_event = ORA.10
		}
		if = {
			limit = {
				has_country_flag = {
					flag = ORA_ref
					value > 2
				}
			}
			country_event = ORA.11
		}

		if = {
			limit = {
				has_country_flag = {
					flag = ORA_rad
					value < 3
				}
				has_country_flag = {
					flag = ORA_mid
					value < 3
				}
				has_country_flag = {
					flag = ORA_ref
					value < 3
				}
			}
			country_event = ORA.12
		}
	}
}
