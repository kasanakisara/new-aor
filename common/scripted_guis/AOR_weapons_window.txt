scripted_gui = {
	AOR_weapons_button = {
		window_name = "AOR_weapons_button"
		context_type = country_context
		visible = {
			always = yes
		}
		parent_window_name = "countrypoliticsview"
		effects = {
			AOR_weapons_button_click ={
				if = {
					limit = {
						NOT = {
							has_country_flag = open_AOR_weapons_button_window
						}
					}
					set_country_flag = open_AOR_weapons_button_window
				}
				else = {
					clr_country_flag = open_AOR_weapons_button_window
				}
			}
		}
	}

	AOR_weapons_window = {
		window_name = "AOR_weapons_window"
		context_type = player_context

		visible = {
			has_country_flag = open_AOR_weapons_button_window
		}
	}
}
