scripted_gui = {
	FRA_ENG_Cold_War_button = {
		window_name = "AOR_FRA_ENG_Cold_War_button"
		context_type = country_context
		visible = {
			always = yes
		}
		parent_window_name = "countrypoliticsview"
		effects = {
			FRA_ENG_Cold_War_button_click ={
				if = {
					limit = {
						NOT = {
							has_country_flag = open_FRA_ENG_Cold_War_button_window
						}
					}
					set_country_flag = open_FRA_ENG_Cold_War_button_window
				}
				else = {
					clr_country_flag = open_FRA_ENG_Cold_War_button_window
				}
			}
		}
	}

	FRA_ENG_Cold_War_window = {
		window_name = "FRA_ENG_Cold_War_window"
		context_type = player_context

		visible = {
			has_country_flag = open_FRA_ENG_Cold_War_button_window
		}
	}
}
