AOR_RUS_modifier_timed_ideas = { #延长内战ideas
	if = {
		limit = { has_idea = SOV_Civil_war_0 }
		modify_timed_idea = { idea = SOV_Civil_war_0 days = 90 }
	}
	if = {
		limit = { has_idea = SOV_Civil_war_stalin_1 }
		modify_timed_idea = { idea = SOV_Civil_war_stalin_1 days = 90 }
	}
	if = {
		limit = { has_idea = SOV_Civil_war_Frunze_1 }
		modify_timed_idea = { idea = SOV_Civil_war_Frunze_1 days = 90 }
	}
}

AOR_RUS_National_industrialization = { #落后的农业国
	if = {
		limit = { has_idea = SOV_Backward_agricultural_country_2 }
		remove_ideas = SOV_Backward_agricultural_country_2
	}
	if = {
		limit = { has_idea = SOV_Backward_agricultural_country_1 }
		swap_ideas = {
			remove_idea = SOV_Backward_agricultural_country_1
			add_idea = SOV_Backward_agricultural_country_2
		}
	}
	if = {
		limit = { has_idea = SOV_Backward_agricultural_country_0 }
		swap_ideas = {
			remove_idea = SOV_Backward_agricultural_country_0
			add_idea = SOV_Backward_agricultural_country_1
		}
	}
}

RUS_give_gun_licences = {
	if = {
		limit = {
			has_tech = infantry_weapons
		}
		if = {
			limit = { has_dlc = "Death or Dishonor" }
			create_production_license = {
				target = PREV
				equipment = {
					type = infantry_equipment_0
				}
				cost_factor = 0
			}
		}
		PREV = {
			add_equipment_to_stockpile = {
				type = infantry_equipment_0
				amount = 200
				producer = ROOT
			}
		}
	}
	
	esle_if = {
		limit = {
			has_tech = infantry_weapons1
		}
		if = {
			limit = { has_dlc = "Death or Dishonor" }
			create_production_license = {
				target = PREV
				equipment = {
					type = infantry_equipment_1
				}
				cost_factor = 0
			}
		}
		PREV = {
			add_equipment_to_stockpile = {
				type = infantry_equipment_1
				amount = 200
				producer = ROOT
			}
		}
	}
}
