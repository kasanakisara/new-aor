#动态修正
SWE_Norweign_Independence_value_check = {
	if = {
		limit = {
			check_variable = { SWE_Norweign_Independence_value = 50 compare = less_than_or_equals }
		}
		set_variable = { SWE_Norweign_Independence_stability_weekly = -0.05 }
		set_variable = { SWE_Norweign_Independence_stability_factor = -0.1 }
		set_variable = { SWE_Norweign_Independence_stability_power_cost = 0.1 }
	}
	else_if = {
		limit = {
			check_variable = { SWE_Norweign_Independence_value = 60 compare = less_than_or_equals }
		}
		set_variable = { SWE_Norweign_Independence_stability_weekly = -0.075 }
		set_variable = { SWE_Norweign_Independence_stability_factor = -0.15 }
		set_variable = { SWE_Norweign_Independence_stability_power_cost = 0.15 }
	}
	else_if = {
		limit = {
			check_variable = { SWE_Norweign_Independence_value = 70 compare = less_than_or_equals }
		}
		set_variable = { SWE_Norweign_Independence_stability_weekly = -0.1 }
		set_variable = { SWE_Norweign_Independence_stability_factor = -0.25 }
		set_variable = { SWE_Norweign_Independence_stability_power_cost = 0.2 }
	}
	else_if = {
		limit = {
			check_variable = { SWE_Norweign_Independence_value = 80 compare = less_than_or_equals }
		}
		set_variable = { SWE_Norweign_Independence_stability_weekly = -0.15 }
		set_variable = { SWE_Norweign_Independence_stability_factor = -0.35 }
		set_variable = { SWE_Norweign_Independence_stability_power_cost = 0.25 }
	}
	else_if = {
		limit = {
			check_variable = { SWE_Norweign_Independence_value = 90 compare = less_than_or_equals }
		}
		set_variable = { SWE_Norweign_Independence_stability_weekly = -0.2 }
		set_variable = { SWE_Norweign_Independence_stability_factor = -0.5 }
		set_variable = { SWE_Norweign_Independence_stability_power_cost = 0.3 }
	}
	else_if = {
		limit = {
			check_variable = { SWE_Norweign_Independence_value < 100 }
		}
		set_variable = { SWE_Norweign_Independence_stability_weekly = -0.3 }
		set_variable = { SWE_Norweign_Independence_stability_factor = -0.65 }
		set_variable = { SWE_Norweign_Independence_stability_power_cost = 0.45 }
	}
}

#马克思主义支持度对挪威独立度修正
SWE_Marxism_effect_Norway_Independence_check = {
	set_temp_variable = { Marxism_num = party_popularity@Marxism }
	if = {
		limit = {
			check_variable = { Marxism_num = 25 compare = less_than_or_equals }
		}
		add_to_variable = { SWE_Norweign_Independence_add = 0.5 }
	}
	else_if = {
		limit = {
			check_variable = { Marxism_num = 35 compare = less_than_or_equals }
		}
		add_to_variable = { SWE_Norweign_Independence_add = 0.75 }
	}
	else_if = {
		limit = {
			check_variable = { Marxism_num = 45 compare = less_than_or_equals }
		}
		add_to_variable = { SWE_Norweign_Independence_add = 1 }
	}
	else_if = {
		limit = {
			check_variable = { Marxism_num = 75 compare = less_than_or_equals }
		}
		add_to_variable = { SWE_Norweign_Independence_add = 1.25 }
	}
	else_if = {
		limit = {
			check_variable = { Marxism_num = 85 compare = less_than_or_equals }
		}
		add_to_variable = { SWE_Norweign_Independence_add = 2 }
	}
	else_if = {
		limit = {

		limit = {
