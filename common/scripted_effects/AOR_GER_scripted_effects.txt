AOR_GER_inflation_clear_all = { #清除通货膨胀系统
	clr_country_flag = {
		AOR_GER_inflation_0
		AOR_GER_inflation_1
		AOR_GER_inflation_2
		AOR_GER_inflation_3
		AOR_GER_inflation_4
		AOR_GER_inflation_5
	}
}

AOR_GER_inflation_level_up = { #德国通货膨胀
	custom_effect_tooltip = AOR_GER_inflation_level_up_tt
	if = {
		limit = { has_country_flag = AOR_GER_inflation_0 }
		clr_country_flag = AOR_GER_inflation_0
		set_country_flag = AOR_GER_inflation_1
	}
	else_if = {
		limit = { has_country_flag = AOR_GER_inflation_1 }
		clr_country_flag = AOR_GER_inflation_1
		set_country_flag = AOR_GER_inflation_2

}
