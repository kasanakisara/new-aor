	name = ENG_Parliament_Icon_of_motion
	text = {
		trigger = {
			check_variable = {
				ENG_Parliament_num_of_motion_going = 1
			}
		}
		localization_key = GFX_ENG_this_is_an_icon
	}
	text = {
		trigger = {
			check_variable = {
				ENG_Parliament_num_of_motion_going = 0
			}
		}
		localization_key = GFX_ENG_this_is_an_icon
	}
}

#通过与否
defined_text = {
	name = ENG_Parliament_support_state_of_motion
	text = {
		trigger = {
			check_variable = {
				ENG_Parliament_motion_going^ENG_Parliament_num_of_motion_going = 1
			}
		}
		localization_key = ENG_Parliament_passage_success
	}
	text = {
		trigger = {
			check_variable = {
				ENG_Parliament_motion_going^ENG_Parliament_num_of_motion_going = -1
			}
		}
		localization_key = ENG_Parliament_passage_fail
	}
	text = {
		trigger = {
			check_variable = {
				ENG_Parliament_motion_going^ENG_Parliament_num_of_motion_going = 0
			}
		}
		localization_key = ENG_Parliament_passage_stalled
	}
}
