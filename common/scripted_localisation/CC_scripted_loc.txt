# Allows you to create your own dynamic keys
# to be called in localization.
# defined_text -> this is it, we're defining the text
# text -> a discrete entry that can be picked to display in loc.
# trigger -> determines if a text entry will be picked or not.
# (The triggers need to be valid for the scope the key is called in
# (eg Root or From.From).)
# localization_key -> points to the localization key
# that'll be used if trigger passes


defined_text = {
	name = CC_luduan_mingcheng
	text = {
		trigger = {
			check_variable = { CC_choose_section = 0 }
		}
		localization_key = CC_no_choose
	}
	text = {
		trigger = {
			check_variable = { CC_choose_section = 1 }
		}
		localization_key = CC_choose_aiji
	}
	text = {
		trigger = {
			check_variable = { CC_choose_section = 2 }
		}
		localization_key = CC_choose_sudan
	}
	text = {
		trigger = {
			check_variable = { CC_choose_section = 3 }
		}
		localization_key = CC_choose_dongfei
	}
	text = {
		trigger = {
			check_variable = { CC_choose_section = 4 }
		}
		localization_key = CC_choose_zhongfei
	}
	text = {
		trigger = {
			check_variable = { CC_choose_section = 5 }
		}
		localization_key = CC_choose_nanfei
	}
}
